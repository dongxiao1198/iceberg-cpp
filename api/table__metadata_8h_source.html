157 using PartitionSpecsMapRef = std::reference_wrapper<const PartitionSpecsMap>;
158 using SortOrdersMapRef = std::reference_wrapper<const SortOrdersMap>;
159 using SnapshotsMapRef = std::reference_wrapper<const SnapshotsMap>;

161 Result<SchemasMapRef> GetSchemasById() const;
162 Result<PartitionSpecsMapRef> GetPartitionSpecsById() const;
163 Result<SortOrdersMapRef> GetSortOrdersById() const;
164 Result<SnapshotsMapRef> GetSnapshotsById() const;
165
166 private:
167 static Result<SchemasMap> InitSchemasMap(const TableMetadata* metadata);
168 static Result<PartitionSpecsMap> InitPartitionSpecsMap(const TableMetadata* metadata);
169 static Result<SortOrdersMap> InitSortOrdersMap(const TableMetadata* metadata);
170 static Result<SnapshotsMap> InitSnapshotMap(const TableMetadata* metadata);
171
172 const TableMetadata* metadata_;
173 Lazy<InitSchemasMap> schemas_map_;
174 Lazy<InitPartitionSpecsMap> partition_specs_map_;
175 Lazy<InitSortOrdersMap> sort_orders_map_;
176 Lazy<InitSnapshotMap> snapshot_map_;
177};
